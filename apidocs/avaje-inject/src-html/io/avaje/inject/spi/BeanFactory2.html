001package io.avaje.inject.spi;
002
003/**
004 * Factory for creating a bean given two request scope arguments.
005 *
006 * @param <T>  The type of bean to create
007 * @param <R>  The first request argument
008 * @param <R2> The second request argument
009 */
010public interface BeanFactory2<T, R, R2> {
011
012  /**
013   * Create and return the bean.
014   */
015  T create(R argument, R2 argument2);
016}
